##Spearman
from scipy.stats import spearmanr
coef, p = spearmanr(Score, GDP)
print('Coeficiente de correlacion de Spearman: %.3f' % coef)
## Interpretar su significancia
alpha= 0.05 # 95% de posibilidad de acierto de que las pruebas no esten relacionadas
if p > alpha:
    print('Las pruebas no estan relacionadas (fallo en rechazar la hipotesis) p= %.3f'% p)
else:
    print('Las pruebas estan relacionadas (rechaza la hipotesis) p= %.3f' % p)
